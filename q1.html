Analyst Snippet (Python)


import pandas as pd

def ebitda_margin(ebitda, revenue):
    return round(100 * ebitda / revenue, 2)

# Example: compute segment margins
segments = pd.DataFrame({
    "segment": ["Cloud", "Payments", "AdTech"],
    "revenue": [420.0, 265.0, 190.0],
    "ebitda": [134.0, 72.5, 41.8]
})

segments["ebitda_margin_%"] = ebitda_margin(segments["ebitda"], segments["revenue"])  # vectorized in pandas
print(segments)
